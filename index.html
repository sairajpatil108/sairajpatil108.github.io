2. Scheduling Algorithms

import java.util.*;

public class SchedulingAlgorithms {
    
    // FCFS Scheduling
    public static void fcfs(int[] burstTimes) {
        int n = burstTimes.length;
        int[] waitingTimes = new int[n];
        int[] turnaroundTimes = new int[n];
        
        waitingTimes[0] = 0;
        turnaroundTimes[0] = burstTimes[0];
        
        for (int i = 1; i < n; i++) {
            waitingTimes[i] = waitingTimes[i-1] + burstTimes[i-1];
            turnaroundTimes[i] = waitingTimes[i] + burstTimes[i];
        }
        
        System.out.println("FCFS Scheduling:");
        printResults(waitingTimes, turnaroundTimes);
    }
    
    // SJF Scheduling
    public static void sjf(int[] burstTimes) {
        int n = burstTimes.length;
        int[] waitingTimes = new int[n];
        int[] turnaroundTimes = new int[n];
        int[] sortedBurst = burstTimes.clone();
        Arrays.sort(sortedBurst);
        
        waitingTimes[0] = 0;
        turnaroundTimes[0] = sortedBurst[0];
        
        for (int i = 1; i < n; i++) {
            waitingTimes[i] = waitingTimes[i-1] + sortedBurst[i-1];
            turnaroundTimes[i] = waitingTimes[i] + sortedBurst[i];
        }
        
        System.out.println("SJF Scheduling:");
        printResults(waitingTimes, turnaroundTimes);
    }
    
    // Round Robin Scheduling
    public static void roundRobin(int[] burstTimes, int quantum) {
        int n = burstTimes.length;
        int[] remainingTime = burstTimes.clone();
        int[] waitingTimes = new int[n];
        int currentTime = 0;
        boolean done;
        
        do {
            done = true;
            for (int i = 0; i < n; i++) {
                if (remainingTime[i] > 0) {
                    done = false;
                    if (remainingTime[i] > quantum) {
                        currentTime += quantum;
                        remainingTime[i] -= quantum;
                    } else {
                        currentTime += remainingTime[i];
                        waitingTimes[i] = currentTime - burstTimes[i];
                        remainingTime[i] = 0;
                    }
                }
            }
        } while (!done);
        
        int[] turnaroundTimes = new int[n];
        for (int i = 0; i < n; i++) {
            turnaroundTimes[i] = burstTimes[i] + waitingTimes[i];
        }
        
        System.out.println("Round Robin Scheduling:");
        printResults(waitingTimes, turnaroundTimes);
    }
    
    // Priority Scheduling
    public static void priority(int[] burstTimes, int[] priorities) {
        int n = burstTimes.length;
        int[] waitingTimes = new int[n];
        int[] turnaroundTimes = new int[n];
        
        // Sort processes by priority (assuming lower number is higher priority)
        Integer[] indices = new Integer[n];
        for (int i = 0; i < n; i++) indices[i] = i;
        Arrays.sort(indices, (a, b) -> priorities[a] - priorities[b]);
        
        waitingTimes[indices[0]] = 0;
        turnaroundTimes[indices[0]] = burstTimes[indices[0]];
        
        for (int i = 1; i < n; i++) {
            waitingTimes[indices[i]] = waitingTimes[indices[i-1]] + burstTimes[indices[i-1]];
            turnaroundTimes[indices[i]]] = waitingTimes[indices[i]] + burstTimes[indices[i]];
        }
        
        System.out.println("Priority Scheduling:");
        printResults(waitingTimes, turnaroundTimes);
    }
    
    private static void printResults(int[] waitingTimes, int[] turnaroundTimes) {
        System.out.println("Process\tWaiting Time\tTurnaround Time");
        for (int i = 0; i < waitingTimes.length; i++) {
            System.out.println((i+1) + "\t" + waitingTimes[i] + "\t\t" + turnaroundTimes[i]);
        }
        System.out.println();
    }
    
    public static void main(String[] args) {
        int[] burstTimes = {10, 5, 8};
        int[] priorities = {2, 1, 3};
        
        fcfs(burstTimes);
        sjf(burstTimes);
        roundRobin(burstTimes, 2);
        priority(burstTimes, priorities);
    }
}

3. IPC using Pipes and Redirection

import java.io.*;

public class IPCExample {
    public static void main(String[] args) throws IOException {
        // Using ProcessBuilder for pipe simulation
        ProcessBuilder pb1 = new ProcessBuilder("echo", "Hello from process 1");
        ProcessBuilder pb2 = new ProcessBuilder("grep", "Hello");
        
        // Redirect output of pb1 to input of pb2
        Process p1 = pb1.start();
        InputStream p1Output = p1.getInputStream();
        pb2.redirectInput(ProcessBuilder.Redirect.PIPE);
        pb2.redirectOutput(ProcessBuilder.Redirect.INHERIT);
        Process p2 = pb2.start();
        
        // Write p1 output to p2 input
        OutputStream p2Input = p2.getOutputStream();
        byte[] buffer = new byte[1024];
        int bytesRead;
        while ((bytesRead = p1Output.read(buffer)) != -1) {
            p2Input.write(buffer, 0, bytesRead);
        }
        p2Input.close();
        
        try {
            p1.waitFor();
            p2.waitFor();
        } catch (InterruptedException e) {
            e.printStackTrace();
        }
        
        System.out.println("Processes completed");
        
        // Simple redirection example
        ProcessBuilder pb = new ProcessBuilder("ls", "-l");
        File outputFile = new File("output.txt");
        pb.redirectOutput(outputFile);
        Process p = pb.start();
        try {
            p.waitFor();
        } catch (InterruptedException e) {
            e.printStackTrace();
        }
        System.out.println("Output redirected to output.txt");
    }
}

4. Reader-Writer Problem

import java.util.concurrent.*;

public class ReaderWriterProblem {
    private static Semaphore readLock = new Semaphore(1);
    private static Semaphore writeLock = new Semaphore(1);
    private static int readCount = 0;
    private static int sharedData = 0;
    
    static class Reader implements Runnable {
        private int id;
        
        public Reader(int id) {
            this.id = id;
        }
        
        @Override
        public void run() {
            try {
                // Acquire read lock
                readLock.acquire();
                readCount++;
                if (readCount == 1) {
                    writeLock.acquire();
                }
                readLock.release();
                
                // Reading
                System.out.println("Reader " + id + " reads: " + sharedData);
                Thread.sleep(100);
                
                // Release read lock
                readLock.acquire();
                readCount--;
                if (readCount == 0) {
                    writeLock.release();
                }
                readLock.release();
            } catch (InterruptedException e) {
                e.printStackTrace();
            }
        }
    }
    
    static class Writer implements Runnable {
        private int id;
        
        public Writer(int id) {
            this.id = id;
        }
        
        @Override
        public void run() {
            try {
                writeLock.acquire();
                
                // Writing
                sharedData++;
                System.out.println("Writer " + id + " writes: " + sharedData);
                Thread.sleep(100);
                
                writeLock.release();
            } catch (InterruptedException e) {
                e.printStackTrace();
            }
        }
    }
    
    public static void main(String[] args) {
        ExecutorService executor = Executors.newFixedThreadPool(5);
        
        // Create readers and writers
        for (int i = 0; i < 3; i++) {
            executor.execute(new Reader(i));
            executor.execute(new Writer(i));
        }
        
        executor.shutdown();
        try {
            executor.awaitTermination(5, TimeUnit.SECONDS);
        } catch (InterruptedException e) {
            e.printStackTrace();
        }
        
        System.out.println("\nWithout synchronization:");
        // Similar code but without semaphores would lead to race conditions
    }
}

5. Banker's Algorithm

import java.util.*;

public class BankersAlgorithm {
    private int[][] need;
    private int[][] allocation;
    private int[] available;
    private int[] work;
    private boolean[] finish;
    private int numProcesses;
    private int numResources;
    
    public BankersAlgorithm(int[][] max, int[][] allocation, int[] available) {
        this.numProcesses = max.length;
        this.numResources = available.length;
        this.allocation = allocation;
        this.available = available;
        this.work = Arrays.copyOf(available, available.length);
        this.finish = new boolean[numProcesses];
        this.need = new int[numProcesses][numResources];
        
        // Calculate need matrix
        for (int i = 0; i < numProcesses; i++) {
            for (int j = 0; j < numResources; j++) {
                need[i][j] = max[i][j] - allocation[i][j];
            }
        }
    }
    
    public boolean isSafe() {
        // Initialize work and finish
        System.arraycopy(available, 0, work, 0, numResources);
        Arrays.fill(finish, false);
        
        int[] safeSequence = new int[numProcesses];
        int count = 0;
        
        while (count < numProcesses) {
            boolean found = false;
            
            for (int i = 0; i < numProcesses; i++) {
                if (!finish[i] && checkNeedLessThanWork(i)) {
                    // Process can complete
                    for (int j = 0; j < numResources; j++) {
                        work[j] += allocation[i][j];
                    }
                    safeSequence[count] = i;
                    finish[i] = true;
                    count++;
                    found = true;
                }
            }
            
            if (!found) {
                // No process found that can complete
                System.out.println("System is not in safe state");
                return false;
            }
        }
        
        System.out.println("System is in safe state");
        System.out.print("Safe sequence: ");
        for (int i = 0; i < numProcesses; i++) {
            System.out.print("P" + safeSequence[i] + " ");
        }
        System.out.println();
        return true;
    }
    
    private boolean checkNeedLessThanWork(int process) {
        for (int i = 0; i < numResources; i++) {
            if (need[process][i] > work[i]) {
                return false;
            }
        }
        return true;
    }
    
    public boolean requestResources(int process, int[] request) {
        // Check if request <= need
        for (int i = 0; i < numResources; i++) {
            if (request[i] > need[process][i]) {
                System.out.println("Error: Process has exceeded maximum claim");
                return false;
            }
        }
        
        // Check if request <= available
        for (int i = 0; i < numResources; i++) {
            if (request[i] > available[i]) {
                System.out.println("Process must wait, resources not available");
                return false;
            }
        }
        
        // Try to allocate
        for (int i = 0; i < numResources; i++) {
            available[i] -= request[i];
            allocation[process][i] += request[i];
            need[process][i] -= request[i];
        }
        
        // Check if new state is safe
        if (isSafe()) {
            return true;
        } else {
            // Rollback
            for (int i = 0; i < numResources; i++) {
                available[i] += request[i];
                allocation[process][i] -= request[i];
                need[process][i] += request[i];
            }
            System.out.println("Request denied - would lead to unsafe state");
            return false;
        }
    }
    
    public static void main(String[] args) {
        // Example from book
        int[][] max = {
            {7, 5, 3},
            {3, 2, 2},
            {9, 0, 2},
            {2, 2, 2},
            {4, 3, 3}
        };
        
        int[][] allocation = {
            {0, 1, 0},
            {2, 0, 0},
            {3, 0, 2},
            {2, 1, 1},
            {0, 0, 2}
        };
        
        int[] available = {3, 3, 2};
        
        BankersAlgorithm banker = new BankersAlgorithm(max, allocation, available);
        banker.isSafe();
        
        // Test resource request
        int[] request = {1, 0, 2};
        System.out.println("\nTesting resource request P1 (1, 0, 2):");
        banker.requestResources(1, request);
    }
}

6. Memory Allocation Techniques

import java.util.*;

public class MemoryAllocation {
    
    // First Fit
    public static void firstFit(int[] blockSizes, int[] processSizes) {
        int m = blockSizes.length;
        int n = processSizes.length;
        int[] allocation = new int[n];
        Arrays.fill(allocation, -1);
        
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < m; j++) {
                if (blockSizes[j] >= processSizes[i]) {
                    allocation[i] = j;
                    blockSizes[j] -= processSizes[i];
                    break;
                }
            }
        }
        
        printAllocation("First Fit", processSizes, allocation, blockSizes);
    }
    
    // Best Fit
    public static void bestFit(int[] blockSizes, int[] processSizes) {
        int m = blockSizes.length;
        int n = processSizes.length;
        int[] allocation = new int[n];
        Arrays.fill(allocation, -1);
        
        for (int i = 0; i < n; i++) {
            int bestIdx = -1;
            for (int j = 0; j < m; j++) {
                if (blockSizes[j] >= processSizes[i]) {
                    if (bestIdx == -1 || blockSizes[j] < blockSizes[bestIdx]) {
                        bestIdx = j;
                    }
                }
            }
            
            if (bestIdx != -1) {
                allocation[i] = bestIdx;
                blockSizes[bestIdx] -= processSizes[i];
            }
        }
        
        printAllocation("Best Fit", processSizes, allocation, blockSizes);
    }
    
    // Next Fit
    public static void nextFit(int[] blockSizes, int[] processSizes) {
        int m = blockSizes.length;
        int n = processSizes.length;
        int[] allocation = new int[n];
        Arrays.fill(allocation, -1);
        int j = 0;
        
        for (int i = 0; i < n; i++) {
            int count = 0;
            while (count < m) {
                if (blockSizes[j] >= processSizes[i]) {
                    allocation[i] = j;
                    blockSizes[j] -= processSizes[i];
                    break;
                }
                j = (j + 1) % m;
                count++;
            }
        }
        
        printAllocation("Next Fit", processSizes, allocation, blockSizes);
    }
    
    // Worst Fit
    public static void worstFit(int[] blockSizes, int[] processSizes) {
        int m = blockSizes.length;
        int n = processSizes.length;
        int[] allocation = new int[n];
        Arrays.fill(allocation, -1);
        
        for (int i = 0; i < n; i++) {
            int worstIdx = -1;
            for (int j = 0; j < m; j++) {
                if (blockSizes[j] >= processSizes[i]) {
                    if (worstIdx == -1 || blockSizes[j] > blockSizes[worstIdx]) {
                        worstIdx = j;
                    }
                }
            }
            
            if (worstIdx != -1) {
                allocation[i] = worstIdx;
                blockSizes[worstIdx] -= processSizes[i];
            }
        }
        
        printAllocation("Worst Fit", processSizes, allocation, blockSizes);
    }
    
    private static void printAllocation(String strategy, int[] processSizes, int[] allocation, int[] blockSizes) {
        System.out.println("\n" + strategy + " Allocation:");
        System.out.println("Process No.\tProcess Size\tBlock No.");
        for (int i = 0; i < allocation.length; i++) {
            System.out.print(" " + (i+1) + "\t\t" + processSizes[i] + "\t\t");
            if (allocation[i] != -1) {
                System.out.print(allocation[i] + 1);
            } else {
                System.out.print("Not Allocated");
            }
            System.out.println();
        }
        
        System.out.println("\nRemaining block sizes:");
        for (int i = 0; i < blockSizes.length; i++) {
            System.out.println("Block " + (i+1) + ": " + blockSizes[i]);
        }
    }
    
    public static void main(String[] args) {
        int[] blockSizes = {100, 500, 200, 300, 600};
        int[] processSizes = {212, 417, 112, 426};
        
        // Make copies for each algorithm since they modify the array
        firstFit(Arrays.copyOf(blockSizes, blockSizes.length), Arrays.copyOf(processSizes, processSizes.length));
        bestFit(Arrays.copyOf(blockSizes, blockSizes.length), Arrays.copyOf(processSizes, processSizes.length));
        nextFit(Arrays.copyOf(blockSizes, blockSizes.length), Arrays.copyOf(processSizes, processSizes.length));
        worstFit(Arrays.copyOf(blockSizes, blockSizes.length), Arrays.copyOf(processSizes, processSizes.length));
    }
}

7. Page Replacement Algorithms

import java.util.*;

public class PageReplacement {
    
    // FIFO Page Replacement
    public static int fifo(int[] pages, int capacity) {
        Queue queue = new LinkedList<>();
        HashSet set = new HashSet<>();
        int pageFaults = 0;
        
        for (int page : pages) {
            if (!set.contains(page)) {
                pageFaults++;
                if (queue.size() == capacity) {
                    int oldest = queue.poll();
                    set.remove(oldest);
                }
                queue.add(page);
                set.add(page);
            }
        }
        
        return pageFaults;
    }
    
    // LRU Page Replacement
    public static int lru(int[] pages, int capacity) {
        HashSet set = new HashSet<>();
        HashMap indexes = new HashMap<>();
        int pageFaults = 0;
        
        for (int i = 0; i < pages.length; i++) {
            int page = pages[i];
            if (!set.contains(page)) {
                pageFaults++;
                if (set.size() == capacity) {
                    int lru = Integer.MAX_VALUE;
                    int val = Integer.MIN_VALUE;
                    
                    for (int p : set) {
                        if (indexes.get(p) < lru) {
                            lru = indexes.get(p);
                            val = p;
                        }
                    }
                    
                    set.remove(val);
                }
                set.add(page);
            }
            indexes.put(page, i);
        }
        
        return pageFaults;
    }
    
    // Optimal Page Replacement
    public static int optimal(int[] pages, int capacity) {
        HashSet set = new HashSet<>();
        int pageFaults = 0;
        
        for (int i = 0; i < pages.length; i++) {
            int page = pages[i];
            if (!set.contains(page)) {
                pageFaults++;
                if (set.size() == capacity) {
                    int farthest = -1;
                    int victim = -1;
                    
                    for (int p : set) {
                        int j;
                        for (j = i + 1; j < pages.length; j++) {
                            if (p == pages[j]) {
                                if (j > farthest) {
                                    farthest = j;
                                    victim = p;
                                }
                                break;
                            }
                        }
                        if (j == pages.length) {
                            victim = p;
                            break;
                        }
                    }
                    
                    set.remove(victim);
                }
                set.add(page);
            }
        }
        
        return pageFaults;
    }
    
    public static void main(String[] args) {
        int[] pages = {7, 0, 1, 2, 0, 3, 0, 4, 2, 3, 0, 3, 2};
        int capacity = 4;
        
        System.out.println("FIFO Page Faults: " + fifo(pages, capacity));
        System.out.println("LRU Page Faults: " + lru(pages, capacity));
        System.out.println("Optimal Page Faults: " + optimal(pages, capacity));
    }
}

8. Disk Scheduling Algorithms

import java.util.*;

public class DiskScheduling {
    
    // FIFO (FCFS)
    public static int fifo(int[] requests, int head) {
        int totalMovement = 0;
        int current = head;
        
        for (int request : requests) {
            totalMovement += Math.abs(request - current);
            current = request;
        }
        
        return totalMovement;
    }
    
    // SSTF (Shortest Seek Time First)
    public static int sstf(int[] requests, int head) {
        List remainingRequests = new ArrayList<>();
        for (int request : requests) {
            remainingRequests.add(request);
        }
        
        int totalMovement = 0;
        int current = head;
        
        while (!remainingRequests.isEmpty()) {
            int closest = findClosest(remainingRequests, current);
            totalMovement += Math.abs(closest - current);
            current = closest;
            remainingRequests.remove(Integer.valueOf(closest));
        }
        
        return totalMovement;
    }
    
    private static int findClosest(List requests, int current) {
        int closest = requests.get(0);
        int minDistance = Math.abs(closest - current);
        
        for (int request : requests) {
            int distance = Math.abs(request - current);
            if (distance < minDistance) {
                minDistance = distance;
                closest = request;
            }
        }
        
        return closest;
    }
    
    // SCAN (Elevator)
    public static int scan(int[] requests, int head, int diskSize, boolean movingRight) {
        List left = new ArrayList<>();
        List right = new ArrayList<>();
        
        for (int request : requests) {
            if (request < head) {
                left.add(request);
            } else {
                right.add(request);
            }
        }
        
        Collections.sort(left);
        Collections.sort(right);
        
        int totalMovement = 0;
        int current = head;
        
        if (movingRight) {
            for (int request : right) {
                totalMovement += request - current;
                current = request;
            }
            totalMovement += diskSize - 1 - current;
            current = diskSize - 1;
            
            for (int i = left.size() - 1; i >= 0; i--) {
                totalMovement += current - left.get(i);
                current = left.get(i);
            }
        } else {
            for (int i = left.size() - 1; i >= 0; i--) {
                totalMovement += current - left.get(i);
                current = left.get(i);
            }
            totalMovement += current;
            current = 0;
            
            for (int request : right) {
                totalMovement += request - current;
                current = request;
            }
        }
        
        return totalMovement;
    }
    
    // C-SCAN (Circular SCAN)
    public static int cscan(int[] requests, int head, int diskSize) {
        List left = new ArrayList<>();
        List right = new ArrayList<>();
        
        for (int request : requests) {
            if (request < head) {
                left.add(request);
            } else {
                right.add(request);
            }
        }
        
        Collections.sort(left);
        Collections.sort(right);
        
        int totalMovement = 0;
        int current = head;
        
        for (int request : right) {
            totalMovement += request - current;
            current = request;
        }
        totalMovement += diskSize - 1 - current;
        totalMovement += diskSize - 1;
        current = 0;
        
        for (int request : left) {
            totalMovement += request - current;
            current = request;
        }
        
        return totalMovement;
    }
    
    public static void main(String[] args) {
        int[] requests = {98, 183, 37, 122, 14, 124, 65, 67};
        int head = 53;
        int diskSize = 200;
        
        System.out.println("FIFO Total Head Movement: " + fifo(requests, head));
        System.out.println("SSTF Total Head Movement: " + sstf(requests, head));
        System.out.println("SCAN Total Head Movement: " + scan(requests, head, diskSize, true));
        System.out.println("C-SCAN Total Head Movement: " + cscan(requests, head, diskSize));
    }
}
